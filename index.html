        const translations = {

            en: {

                winCountPrefix: 'Windows: '

            }

        let currentLang = 'ru';


        function buildCheckerboard() {

            const board = document.getElementById('checkerboard');

            const w = window.innerWidth;

            const h = window.innerHeight;

            const cellSize = Math.min(w / 5, h / 3);

            board.style.width = (cellSize * 5) + 'px';

            board.style.height = (cellSize * 3) + 'px';

            board.innerHTML = '';

            for (let r = 0; r < 3; r++) {

                for (let c = 0; c < 5; c++) {

                    const cell = document.createElement('div');

                    cell.classList.add('cell');

                    cell.classList.add((r + c) % 2 === 0 ? 'white' : 'black');

                    board.appendChild(cell);

                }

            }

        }


        function injectScreenshots() {

            const container = document.getElementById('screens-row');

            container.innerHTML = '';

            for (let i = 1; i <= SCREENS_COUNT; i++) {

                const img = document.createElement('img');

                img.src = `img/screen_${i}.png`;

                img.alt = `Скриншот ${i}`;

                container.appendChild(img);

            }

        }


        async function updateCounts() {

            try {

                const [macRes, winRes] = await Promise.all([

                    fetch(COUNTER_MAC_GET).then(r => r.json()),

                    fetch(COUNTER_WIN_GET).then(r => r.json())

                ]);

                const t = translations[currentLang];

                document.getElementById('mac-count').textContent = t.macCountPrefix + (macRes.value || 0);

                document.getElementById('win-count').textContent = t.winCountPrefix + (winRes.value || 0);

            } catch (e) {

                console.error('Counter error:', e);

                document.getElementById('mac-count').textContent = 'Mac: ?';

                document.getElementById('win-count').textContent = 'Windows: ?';

            }

        }


        document.getElementById('btn-mac').addEventListener('click', async (e) => {

            e.preventDefault();

            try { await fetch(COUNTER_MAC_HIT); } catch (err) {}

            setTimeout(async () => {

                await updateCounts();

                window.location.href = MAC_DOWNLOAD_URL;

            }, 200);

        });


        function setLanguage(lang) {

            currentLang = lang;

            const t = translations[lang];

            document.getElementById('desc-text').textContent = t.desc;

            document.getElementById('btn-mac').textContent = t.btnMac;

            document.getElementById('btn-win').textContent = t.btnWin;

            document.getElementById('support-text').textContent = t.support;

            document.getElementById('btn-ru').classList.toggle('active', lang === 'ru');

            document.getElementById('btn-en').classList.toggle('active', lang === 'en');

            updateCounts();

        }


        document.getElementById('btn-ru').addEventListener('click', () => setLanguage('ru'));

        document.getElementById('btn-en').addEventListener('click', () => setLanguage('en'));


        window.addEventListener('resize', () => requestAnimationFrame(buildCheckerboard));


        window.matchMedia('(prefers-color-scheme: light)').addEventListener('change', () => {

            buildCheckerboard();

        });
